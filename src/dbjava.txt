CREATE DATABASE dbjava;

use dbjava

//tabela cliente
CREATE TABLE cliente (
  idcliente int,
  cpf varchar(11) NOT NULL,
  nome varchar(50),
  sobrenome varchar(150),
  rg varchar(12),
  endereco varchar(150),
  PRIMARY KEY (idcliente));

//tabela veiculo
CREATE TABLE veiculo (
codveiculo int NOT NULL AUTO_INCREMENT,
marca int,
estado int,
locacao int,
categoria int,
valordecompra float,
idade int,
placa varchar(10),
ano int(4),
PRIMARY KEY (codveiculo));

//tabela de locacao
CREATE TABLE loca (
codlocacao int,
dias int,
valor double,
data date,
codcliente int,
codveiculo int,
PRIMARY KEY (codlocacao),
FOREIGN KEY (codcliente) REFERENCES cliente(idcliente),
FOREIGN KEY (codveiculo) REFERENCES veiculo(codveiculo));
